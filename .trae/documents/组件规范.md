## 四、组件API设计

### 4.1 Props定义

```vue
<script setup>
// 定义Props
const props = defineProps({
  // 基础类型
  title: String,

  // 带默认值
  status: {
    type: String,
    default: 'idle'
  },

  // 复杂类型
  data: {
    type: Array,
    default: () => []
  },

  // 必填
  alarmId: {
    type: String,
    required: true
  },

  // 自定义验证
  level: {
    type: Number,
    validator: (v) => v >= 0 && v <= 5
  }
})
</script>
```

### 4.2 Events定义

```vue
<script setup>
// 定义Emits
const emit = defineEmits({
  // 简单事件
  update: null,

  // 带参数验证
  click: (payload) => {
    if (!payload.id) {
      console.warn('click event requires id')
      return false
    }
    return true
  },

  // 带默认值
  change: (value) => {
    return value !== undefined
  }
})

// 触发事件
function handleClick() {
  emit('click', { id: props.alarmId })
}
</script>
```

### 4.3 Slots定义

```vue
<template>
  <div class="alarm-panel">
    <!-- 默认slot -->
    <slot />

    <!-- 命名slot -->
    <slot name="header" />
    <slot name="footer" />

    <!-- 作用域slot -->
    <slot name="item" :item="item" />
  </div>
</template>
```

### 4.4 Expose定义

```vue
<script setup">
// 暴露方法给父组件
defineExpose({
  refresh,
  reset,
  getData
})

function refresh() { /* ... */ }
function reset() { /* ... */ }
function getData() { return data }
</script>
```

### 5.3 RealtimeChart.vue

```vue
<!--
  RealtimeChart - 实时曲线图

  基于ECharts的实时数据图表

  Props:
    - data: Array<{time, value}>
    - color: string
    - height: number

  Events:
    - zoom: 缩放事件
-->
<template>
  <div class="realtime-chart" :style="{ height: `${height}px` }">
    <div ref="chartRef" class="chart-container" />
  </div>
</template>

<script setup>
import { ref, onMounted, onUnmounted, watch } from 'vue'
import * as echarts from 'echarts'

const props = defineProps({
  data: {
    type: Array,
    default: () => []
  },
  color: {
    type: String,
    default: '#1890ff'
  },
  height: {
    type: Number,
    default: 300
  }
})

const chartRef = ref(null)
let chart = null

onMounted(() => {
  chart = echarts.init(chartRef.value)
  updateChart()
})

onUnmounted(() => {
  chart?.dispose()
})

watch(() => props.data, updateChart, { deep: true })

function updateChart() {
  if (!chart) return
  chart.setOption({
    xAxis: { type: 'time' },
    yAxis: { type: 'value' },
    series: [{
      type: 'line',
      data: props.data,
      smooth: true,
      lineStyle: { color: props.color }
    }]
  })
}
</script>
```

## 六、组件复用模式

### 6.1 容器-展示分离

```vue
<!-- 容器组件：处理逻辑 -->
<script setup>
// ContainerComponent.vue
import { useAlarmStore } from '@/stores/alarmStore'
import AlarmList from './AlarmList.vue'

const alarmStore = useAlarmStore()

function handleAck(id) {
  alarmStore.acknowledge(id)
}
</script>

<template>
  <AlarmList
    :alarms="alarmStore.activeAlarms"
    @acknowledge="handleAck"
  />
</template>

<!-- 展示组件：只处理显示 -->
<script setup>
// AlarmList.vue
defineProps({
  alarms: Array
})

defineEmits(['acknowledge'])
</script>

<template>
  <div class="alarm-list">
    <div
      v-for="alarm in alarms"
      :key="alarm.id"
      @click="$emit('acknowledge', alarm.id)"
    >
      {{ alarm.message }}
    </div>
  </div>
</template>
```

### 6.2 插槽模式

```vue
<!-- 通用卡片组件 -->
```


### 8.3 组件销毁清理

```vue
<script setup>
import { ref, onMounted, onUnmounted } from 'vue'

const timer = ref(null)

onMounted(() => {
  timer.value = setInterval(() => {
    // ...
  }, 1000)

  if (timer.value) {
    clearInterval(timer.value)
  }
```
